===FreeBSD开启SSH远程登录===
1、安装时选择上SSH，或者源码安装SSH
2、使用root登陆系统
3、ee /etc/inetd.conf    #去掉ssh前的#，按ctrl+c，再输入exit保存退出
4、ee /etc/rc.conf         #添加一行sshd_enable="YES"
5、ee /etc/ssh/sshd_config
#PermitRootLogin no改为PermitRootLogin yes                         #允许root登陆
#PasswordAuthentication no改为PasswordAuthenticationyes   #使用系统PAM认证
#PermitEmptyPasswords no改为PermitEmptyPasswords no      #不允许空密码
ee /etc/ssh/sshd_config   #修改端口号
#port=22  #改为 Port=number（你想修改的数字）保存退出
6、ee   /etc/rc.d/sshd start    #启动SSHD服务，
7、netstat -an   #查看服务是否启动，如果看到22端口有监听，恭喜！！！

===TouchPAD===
#To enable psm(4) Synaptics support, add the following line to /boot/loader.conf
hw.psm.synaptics_support="1"
/etc/rc.conf     moused_enable="YES"
/etc/X11/xorg.conf
Section "InputDevice"
    Identifier      "Mouse0"
    Driver          "mouse"
    Option          "Protocol"      "auto"
    Option          "Device"        "/dev/sysmouse"
    Option          "ZAxisMapping"  "4 5 6 7"
EndSection
#Horizontal scrolling is disabled by default. To enable it, add the following to /etc/sysctl.conf
hw.psm.synaptics.vscroll_hor_area=1300
